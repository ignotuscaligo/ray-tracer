# The imgui Conan package ships the platform/renderer backend sources
# (imgui_impl_glfw.*, imgui_impl_opengl3.*) under res/bindings rather than
# compiling them into the library, because the choice of backend is up to the
# consumer. Copy them into editor/imgui_backends/ so the editor target can
# compile exactly the glfw + opengl3 pair it needs.
[imports]
./res/bindings, imgui_impl_glfw.* -> ../editor/imgui_backends
./res/bindings, imgui_impl_opengl3.* -> ../editor/imgui_backends
